\section{Постановка задачи}
\label{sec:problem}
\noindent Дана функция
\begin{equation}
    f:\mathbb{R}^{n} \rightarrow \mathbb{R}
\end{equation}

\noindent Найти
\begin{equation}
    x_n \in B(\epsilon, x_*):  f(x_*) = \min_{x \in \mathbb{R}^{n}} f(x)
\end{equation}
\noindent где $\epsilon > 0$ - заданная точность.
